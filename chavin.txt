git init // inicia un repositorio
git add . // agrega los cambios al repositoio
git commit -m 'comentario'  // guarda los cambios agregados

git log-- oneline // muestra las versiones con resumen de información
git log  // muestra las versiones con información extensa

git checkout id o master // recupera alguna de las versiones en el repositorio

git status // muestra el estatus de los archivos

git reset -- id o master hard // borra las versiones posteriroes al id

git branch // muestra las ramas
